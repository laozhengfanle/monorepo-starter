## 第一次 fork：清理与重命名

### 1. 改包名

把所有 `package.json` 的 `name` 字段改成你的项目命名空间：

```bash
# 仓库根
"name": "my-project"

# apps/server
"name": "@my-org/my-project-server"

# apps/admin
"name": "@my-org/my-project-admin"

# apps/web
"name": "@my-org/my-project-web"

# packages/shared
"name": "@my-org/my-project-shared"

# packages/config
"name": "@my-org/my-project-config"
```

> **不要保留 `monorepo-starter` / `@apps/*` / `@packages/*`**——升基座时会冲突。

### 2. 删你不需要的部分

```bash
# 只做管理后台，没有 C 端
rm -rf apps/web

# 只做 C 端，没有管理后台
rm -rf apps/admin

# 不需要会员等级体系
rm -rf apps/server/src/modules/member
# 同时把 admin 里的 member 引用、seed 里的 member 数据删掉
```

### 3. 改 LICENSE 标识

`LICENSE` 文件里 `Copyright (c) 2026 MonoKit Contributors` 改成你自己的。

### 4. 改 README

`README.md` 顶部品牌名 / 截图 / 链接全部换成你的项目。

### 5. 验证

```bash
pnpm install
docker compose up -d
pnpm -F @apps/server db:migrate
pnpm -F @apps/server db:seed
pnpm dev
```

跑通后 fork 清理完成。

## 加一个新业务模块（CRUD）

举例：加一个「文章」模块，字段 `title / content / author / status`，仅管理员可操作。

### Step 1：数据库

编辑 `apps/server/prisma/schema.prisma`：

```prisma
model Article {
  id        String    @id @db.Uuid
  title     String    @db.VarChar(200)
  content   String    @db.Text
  authorId  String    @map("author_id") @db.Uuid
  status    String    @default("draft") @db.VarChar(20)  // draft / published
  createdAt DateTime  @default(now()) @map("created_at") @db.Timestamptz
  updatedAt DateTime  @updatedAt @map("updated_at") @db.Timestamptz
  deletedAt DateTime? @map("deleted_at") @db.Timestamptz

  author Account @relation(fields: [authorId], references: [id], onDelete: Restrict)

  @@index([authorId])
  @@index([status])
  @@map("article")
}
```

> `Account` 模型需要加反向关系（`Article[]`），不然 Prisma 会报 missing opposite relation。

生成迁移：

```bash
pnpm -F @apps/server db:migrate:create -- --name add_article
# 生成的 SQL 文件会出现在 prisma/migrations/ 下，名字带时间戳
# 务必检查 SQL 是否符合预期，再 commit
```

### Step 2：Zod schema（前后端共享）

新建 `packages/shared/src/schemas/article/article.schema.ts`：

```ts
import { z } from 'zod';

export const ArticleStatusSchema = z.enum(['draft', 'published']);

export const CreateArticleInputSchema = z.object({
    title: z.string().min(1).max(200),
    content: z.string().min(1),
    status: ArticleStatusSchema.default('draft'),
});

export const UpdateArticleInputSchema = CreateArticleInputSchema.partial();

export const ArticleQueryInputSchema = z.object({
    keyword: z.string().optional(),
    status: ArticleStatusSchema.optional(),
    page: z.number().int().min(1).default(1),
    pageSize: z.number().int().min(1).max(100).default(20),
});

export type CreateArticleInput = z.infer<typeof CreateArticleInputSchema>;
export type UpdateArticleInput = z.infer<typeof UpdateArticleInputSchema>;
export type ArticleQueryInput = z.infer<typeof ArticleQueryInputSchema>;
```

在 `packages/shared/src/index.ts` 导出：

```ts
export * from './schemas/article/article.schema.js';
```

重新构建：

```bash
pnpm -F @packages/shared build
```

### Step 3：后端模块

新建 `apps/server/src/modules/article/article.module.ts`：

```ts
import { Module } from '@nestjs/common';
import { ArticleResolver } from './article.resolver.js';
import { ArticleService } from './article.service.js';

@Module({
    providers: [ArticleResolver, ArticleService],
    exports: [ArticleService],
})
export class ArticleModule {}
```

新建 `apps/server/src/modules/article/article.service.ts`：

```ts
import { Injectable, NotFoundException } from '@nestjs/common';
import type { Prisma } from '@prisma/client';
import { PrismaService } from '@/common/prisma/prisma.service.js';
import { uuidv7 } from '@/common/utils/uuid.js';
import type { CreateArticleInput, UpdateArticleInput, ArticleQueryInput } from '@packages/shared';

@Injectable()
export class ArticleService {
    constructor(private prisma: PrismaService) {}

    async list(query: ArticleQueryInput) {
        const where: Prisma.ArticleWhereInput = { deletedAt: null };
        if (query.keyword) where.title = { contains: query.keyword, mode: 'insensitive' };
        if (query.status) where.status = query.status;

        const [items, total] = await this.prisma.$transaction([
            this.prisma.article.findMany({
                where,
                skip: (query.page - 1) * query.pageSize,
                take: query.pageSize,
                orderBy: { createdAt: 'desc' },
            }),
            this.prisma.article.count({ where }),
        ]);
        return { items, total, page: query.page, pageSize: query.pageSize };
    }

    async findById(id: string) {
        const article = await this.prisma.article.findFirst({ where: { id, deletedAt: null } });
        if (!article) throw new NotFoundException('文章不存在');
        return article;
    }

    create(authorId: string, input: CreateArticleInput) {
        return this.prisma.article.create({
            data: { ...input, authorId, id: uuidv7() },
        });
    }

    async update(id: string, input: UpdateArticleInput) {
        await this.findById(id);
        return this.prisma.article.update({ where: { id }, data: input });
    }

    async softDelete(id: string) {
        await this.findById(id);
        return this.prisma.article.update({ where: { id }, data: { deletedAt: new Date() } });
    }
}
```

> **UUID 用 `uuidv7()` 而不是 `crypto.randomUUID()`** —— 项目统一用 UUID v7（时间有序），用 [apps/server/src/common/utils/uuid.ts](../apps/server/src/common/utils/uuid.ts) 的工具函数。

新建 `apps/server/src/modules/article/article.resolver.ts`：

```ts
import { Resolver, Query, Mutation, Args, ID } from '@nestjs/graphql';
import { UseGuards, Req } from '@nestjs/common';
import type { Request } from 'express';
import { ArticleService } from './article.service.js';
import { JwtAuthGuard } from '@/common/guards/jwt-auth.guard.js';
import { AdminPermissionGuard } from '@/common/guards/admin-permission.guard.js';
import { Permission } from '@/common/decorators';
import { ZodArgs } from '@/common/pipes/zod-args.pipe.js';
import { CreateArticleInputSchema, UpdateArticleInputSchema, ArticleQueryInputSchema } from '@packages/shared';

@Resolver()
@UseGuards(JwtAuthGuard, AdminPermissionGuard)
export class ArticleResolver {
    constructor(private service: ArticleService) {}

    @Query()
    @Permission('article:list')
    articles(@ZodArgs('query', ArticleQueryInputSchema) query: any) {
        return this.service.list(query);
    }

    @Query()
    @Permission('article:list')
    article(@Args('id', { type: () => ID }) id: string) {
        return this.service.findById(id);
    }

    @Mutation()
    @Permission('article:create')
    createArticle(
        @Req() req: Request & { user: { accountId: string } },
        @ZodArgs('input', CreateArticleInputSchema) input: any,
    ) {
        return this.service.create(req.user.accountId, input);
    }

    @Mutation()
    @Permission('article:update')
    updateArticle(
        @Args('id', { type: () => ID }) id: string,
        @ZodArgs('input', UpdateArticleInputSchema) input: any,
    ) {
        return this.service.update(id, input);
    }

    @Mutation()
    @Permission('article:delete')
    deleteArticle(@Args('id', { type: () => ID }) id: string) {
        return this.service.softDelete(id);
    }
}
```

> **当前账户 ID 怎么拿**：`JwtAuthGuard` 把 `accountId` 塞到 `req.user.accountId`，所以用 `@Req() req: Request` 取。**没有 `CurrentAccountId` 装饰器**——参考 [auth.controller.ts](../apps/server/src/modules/auth/auth.controller.ts) 现有写法。

```ts
import { gqlClient } from '@/shared/request/graphql-client';

export const articleList = (input: any) =>
    gqlClient.request(/* GraphQL */ `
        query Articles($input: ArticleQueryInput!) {
            articles(input: $input) { items { id title status } total }
        }
    `, { input });

export const deleteArticle = (id: string) =>

            deleteArticle(id: $id)
        }
    `, { id });
```

## 改数据库表

> **永远不要改 `prisma/migrations/` 下已存在的迁移文件**——会破坏已有数据库。

正确流程：

```bash
# 1. 改 schema.prisma
vim apps/server/prisma/schema.prisma

# 2. 生成新迁移（不要自动应用，先看 SQL）
pnpm -F @apps/server db:migrate:create -- --name your_change_name
# 生成的 SQL 在 prisma/migrations/<时间戳>_your_change_name/migration.sql

# 3. 检查 SQL（生产环境慎用 drop / truncate）

# 4. 应用
pnpm -F @apps/server db:migrate
```

如果是「纯字段加」，Prisma 会自动生成安全的 ALTER TABLE。
如果是「重命名 / 删除」，**生产环境先做数据备份**（见 [06-部署运维.md](./06-部署运维.md) 的备份章节）。

## 加 GraphQL 端点

按上面「加一个新业务模块」Step 3 的模式：在 `*.resolver.ts` 加 `@Query` / `@Mutation` 即可。

**注意事项**：

- 每次改 resolver 后跑 `pnpm -F @apps/server generate:schema`，CI 会校验 `schema.gql` artifact 是否同步
- 输入参数用 Zod schema（`@ZodArgs` 装饰器 + 共享 schema），不要直接用 `@Args()` 写内联类型
- 涉及权限的端点必须加 `@Permission('xxx')`

---

## 加 REST 端点

REST 只用于「命令式操作」（认证、上传、健康检查）。**CRUD 全部走 GraphQL**。

加 REST 的位置：`apps/server/src/bff/<context>/<resource>/<resource>.controller.ts`，例如：

- `bff/admin/uploads/upload.controller.ts` —— 管理端上传
- `bff/public/auth/auth.controller.ts` —— 公开认证端点

加完记得用 `@Public()` / `@Permission()` 装饰器标注访问控制。

## 加新的权限码

1. 在 `seed.ts` 的菜单里给对应菜单项加 `permissionCode: 'article:export'`
2. 在 resolver / controller 用 `@Permission('article:export')`
3. 前端用 `v-permission="'article:export'"` 守卫按钮
4. 给角色授权：管理后台「角色管理」勾选对应菜单

> **多权限码 OR 语义**：同一端点支持多权限码时，把多个权限码作为 rest 参数传 `@Permission('dashboard:welcome', 'dashboard:analytics')` —— 拥有任一权限码即可通过。详见 [apps/server/src/common/decorators/permission.decorator.ts](../apps/server/src/common/decorators/permission.decorator.ts)。

## 加第三方 provider（短信 / 邮件 / OAuth / 存储）

四类 provider 都是「接口 + 工厂」模式，加新 provider 步骤相同：

**例：加一个腾讯云短信 provider**

1. 复制 `apps/server/src/common/sms/providers/mock.provider.ts` → `tencent.provider.ts`
2. 实现 `SmsProvider` 接口（`send(phone, template, params)` 方法）
3. 在 `sms.module.ts` 的 providers 数组注册
4. 改 `sms.service.ts` 的 driver 切换逻辑（switch case）
5. 跑 `pnpm -F @apps/server db:seed` 把 driver 选项写进 `system_config`
6. 管理后台「短信服务」选 Tencent → 填 AccessKey

OAuth / Email / Storage 流程相同，接口文件：

- OAuth：`apps/server/src/common/oauth/oauth.provider.ts`
- Email：`apps/server/src/common/email/email.provider.ts`
- Storage：`apps/server/src/common/storage/storage.interface.ts`

## 前端加一个新页面

**快速路径**：

1. 在 `apps/admin/src/features/<domain>/` 新建 `<PageName>.vue`
2. 在 `apps/admin/src/app/router/routes/modules/<domain>.ts` 加路由配置
3. 在 `apps/admin/src/app/router/routes/index.ts` 的 `moduleRoutes` 注册
4. 后端有对应菜单/权限码的话，同步在 `seed.ts` 加菜单
5. 跑 `pnpm -F @apps/server db:seed`（让菜单树包含新页面）

**前端约定**：

- 列表页：用 `<SearchGrid>` + `<ListDisplayFilter>` + `<NDataTable>`
- 表单：用 `useFormSubmit` + `FieldRulePopover`（一字段一规则）
- 按钮权限：用 `v-permission`
- API 客户端：每个业务模块一个 `apps/admin/src/api/<module>.ts`

## 升级基座

> **基座升级是高风险操作**，必须按下面流程做。

### 1. 准备工作

```bash
# 备份当前数据库
docker exec monorepo-postgres pg_dump -U root mono_dev > backup-$(date +%Y%m%d).sql

# 确认当前基座版本
git tag | sort -V | tail -5
```

### 2. 拉取基座更新

```bash
# 添加基座仓库为 remote（首次）
git remote add upstream <monorepo-repo-url>

# 拉取新版本
git fetch upstream
git merge upstream/<version-tag>
```

### 3. 解决冲突

冲突通常发生在两个地方：

- `apps/server/prisma/schema.prisma`：基座可能加了新表或字段
  - **保留你的业务字段 + 接受基座的新字段**
- `apps/server/prisma/seed.ts`：基座可能改了默认数据
  - **接受基座的默认数据 + 保留你的业务菜单**

冲突较少的区域（**这些尽量接受基座更新，不要保留本地**）：

- `apps/server/src/common/`：基础设施
- `apps/admin/src/shared/`、`apps/admin/src/app/`：框架层
- `apps/*/src/api/`：基座的 API 客户端

### 4. 跑迁移 + 验证

```bash
# 安装新依赖
pnpm install

# 跑新迁移（基座可能加了新表）
pnpm -F @apps/server db:migrate

# 重跑 seed（幂等）
pnpm -F @apps/server db:seed

# 跑 typecheck / lint / test / build 全套闸门
pnpm typecheck
pnpm lint
pnpm test
pnpm build

# 重启 dev，手动测核心路径
pnpm dev
```

### 5. 看 Changelog

升级前**必看** [08-Changelog.md](./08-Changelog.md) —— 里面有「破坏性变更」「需要手动操作」的清单。

## 不要做的事

| 反模式 | 后果 |
|---|---|
| 改 `apps/server/src/main.ts` 启动顺序 | 破坏 Helmet / CSRF / 优雅关闭 |
| 改 `apps/server/src/app.module.ts` 的 imports 顺序 | Logger 拿不到，日志乱 |
| 改 `prisma/migrations/` 已存在文件 | 数据库 schema drift |
| 删 `super_admin` 角色 | 默认账号 `root` 变成无角色 |
| 在前端 `.env` 放密钥 | 任何人可读 |
| 把业务代码写进 `common/` / `shared/` | 升基座时会被覆盖 |
| 跳过 `pnpm -F @apps/server generate:schema` | CI 失败，codegen 不一致 |
